#install livewire:
composer require livewire/livewire

#Publishing the configuration file:
php artisan livewire:publish --config   //je config/livewire.php created

#create template layout:
php artisan livewire:layout  // ye command laravel v10 nd php v8+ me use hoga

#creating components view file and class file:
php artisan make:livewire ShowHome

#create table migrate and model and seeder
php artisan make:model Service -ms  //model name any create 
php artisan migrate:fresh --seed

#component create view file:
php artisan make:component TeamCard --view  //TeamCard name ma any name view file

#seeder create:
php artisan make:seeder MemeberTableSeeder  //any name 
run: php artisan db:seed --class=MemeberTableSeeder
